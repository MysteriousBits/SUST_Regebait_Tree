\codeWithDescription{Dinic}{flow_matching/dinic.cpp}
[Flow algorithm with complexity $O(VE\log U)$ where $U = \max |\text{cap}|$. $O(\min(E^{1/2}, V^{2/3})E)$ if $U = 1$; $O(\sqrt{V}E)$ for bipartite matching. \\
\textbf{Time:} {$O(VE\log U)$}]

\codeWithDescription{HopcroftKarp}{flow_matching/hopcroft.cpp}
[Fast bipartite matching. \texttt{g} is the adjacency list of the left partition. \texttt{r} is a vector of -1s of size equal to the right partition; after the call, \texttt{r[i]} is the left-side match of right vertex $i$, or -1 if unmatched. Returns the size of the maximum matching. \\
\textbf{Time:} {$O(E\sqrt{V})$}]
